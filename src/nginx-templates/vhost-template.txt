server {
    server_name DOMAIN;

    listen 80;
    listen 443 ssl;
    HTTP_VERSION_DIRECTIVE
    
    # SSL Configuration
    ssl_certificate /var/www/nginxConfig/certificates/live/DOMAIN/fullchain.pem;
    ssl_certificate_key /var/www/nginxConfig/certificates/live/DOMAIN/privkey.pem;
    ssl_protocols TLS_VERSION;
    
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;
    ssl_prefer_server_ciphers off;
    ssl_ciphers "ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384:ECDHE-ECDSA-CHACHA20-POLY1305:ECDHE-RSA-CHACHA20-POLY1305:DHE-RSA-AES128-GCM-SHA256:DHE-RSA-AES256-GCM-SHA384";
    
    ssl_dhparam /nginx-data/dhparam.pem;

    # Client settings
    client_max_body_size CLIENT_MAX_BODY_SIZE;

    # Logging
    access_log /var/log/nginx/DOMAIN-access.log;
    error_log /var/log/nginx/DOMAIN-error.log;

    # Gzip compression
    GZIP_CONFIG

    BLOCK_EXPLOITS

    # Let's Encrypt challenge
    location /.well-known/acme-challenge/ {
        proxy_pass http://webui/.well-known/acme-challenge/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    CACHE_CONFIG

    location / {
        # Proxy headers
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header Host $host;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        # Proxy timeouts
        proxy_connect_timeout PROXY_TIMEOUTs;
        proxy_send_timeout PROXY_TIMEOUTs;
        proxy_read_timeout PROXY_TIMEOUTs;
        
        # Proxy buffering
        proxy_buffering on;
        proxy_buffer_size PROXY_BUFFER_SIZE;
        proxy_buffers 8 PROXY_BUFFER_SIZE;
        
        # WebSocket support
        WEBSOCKET_CONFIG
        
        # Custom headers
        CUSTOM_HEADERS
        
        # Proxy pass
        proxy_pass http://DESTINATIONIP:DESTINATION_PORT;
    }
    
    # Custom configuration
    CUSTOM_CONFIG
}
